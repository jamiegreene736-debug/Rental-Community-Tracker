<span class="kw">function</span> getNightlyRate(communityName, bedrooms, yearMonth) {
  <span class="kw">const</span> entry = BASE_RATES.find(r => r.community === communityName);
  <span class="kw">const</span> key = bedrooms + <span class="str">"BR"</span>;
  <span class="kw">const</span> base = entry?.[key] ?? (FALLBACK_RATE_PER_BEDROOM * <span class="kw">parseInt</span>(bedrooms));
  <span class="kw">const</span> seasonMap = entry?.region === <span class="str">"florida"</span> ? FLORIDA_SEASONS : HAWAII_SEASONS;
  <span class="kw">const</span> season = seasonMap[yearMonth] ?? <span class="str">"MEDIUM"</span>;
  <span class="kw">const</span> multiplier = SEASON_MULTIPLIERS[entry?.region ?? <span class="str">"hawaii"</span>][season];
  <span class="kw">return</span> Math.round(base * multiplier);
}
